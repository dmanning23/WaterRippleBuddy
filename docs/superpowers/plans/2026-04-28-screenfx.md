# ScreenFX Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build a standalone MonoGame post-processing library (`ScreenFX`) with 7 simultaneous screen-space effects (force ripple, gravity wave, screen shake, chromatic aberration, heat haze, hit flash, anime super) using a per-type ping-pong render target chain.

**Architecture:** Two shared `RenderTarget2D` buffers ping-pong as chain passes run — each effect type handles all its active instances in one GPU pass. Overlay effects (HitFlash, AnimeSuper) draw additively on top of the back buffer after the chain.

**Tech Stack:** MonoGame 3.8 DesktopGL, .NET 8, HLSL ps_3_0/vs_3_0 + OpenGL compat block, MonoGame Content Pipeline (MGCB).

## Critical Shader Authoring Rules

Every `.fx` file in this project MUST follow this exact pattern. Deviating from it causes a black screen on DesktopGL with no error. This pattern was proven the hard way — trust it completely.

```hlsl
#if OPENGL
    #define SV_POSITION POSITION
    #define VS_SHADERMODEL vs_3_0
    #define PS_SHADERMODEL ps_3_0
#else
    #define VS_SHADERMODEL vs_4_0_level_9_1
    #define PS_SHADERMODEL ps_4_0_level_9_1
#endif

// ... uniforms and sampler ...

struct VertexShaderOutput
{
    float4 Position : SV_POSITION;
    float4 Color    : COLOR0;
    float2 TexCoord : TEXCOORD0;
};

float4 PS(VertexShaderOutput input) : COLOR   // ← must be COLOR, not COLOR0 or SV_Target
{
    // ... pixel shader logic ...
}

technique EffectName
{
    pass P0
    {
        PixelShader = compile PS_SHADERMODEL PS();
        // NO VertexShader — SpriteBatch provides it. Adding one will cause conflicts.
    }
}
```

**Three rules:**
1. Always include the `#if OPENGL` block at the top.
2. The technique specifies `PixelShader` only — no `VertexShader`.
3. The pixel shader return semantic is `: COLOR` (not `: COLOR0`, not `: SV_Target`).

```csharp
using Microsoft.Xna.Framework;
using Microsoft.Xna.Framework.Content;
using Microsoft.Xna.Framework.Graphics;
using System.Collections.Generic;

namespace ScreenFX.Effects;

public abstract class EffectLayerBase : IEffectLayer
{
    protected struct EffectInstance
    {
        public Vector2 Position;
        public float Age;
    }

    protected List<EffectInstance> Instances { get; } = new();

    // Layers that don't use the instance list override this (e.g. ScreenShakeLayer).
    public virtual bool IsActive => Instances.Count > 0;

    // Override to true for additive overlay layers (HitFlash, AnimeSuper).
    public virtual bool IsOverlay => false;

    // Adds a new instance, evicting the oldest if at capacity.
    protected void AddInstance(Vector2 position, int maxCount = 16)
    {
        if (Instances.Count >= maxCount) Instances.RemoveAt(0);
        Instances.Add(new EffectInstance { Position = position, Age = 0f });
    }

    // Advances all instances by deltaSeconds; removes those that have reached lifetime.
    protected void TickInstances(float deltaSeconds, float lifetime)
    {
        for (int i = Instances.Count - 1; i >= 0; i--)
        {
            var inst = Instances[i];
            inst.Age += deltaSeconds;
            if (inst.Age >= lifetime)
                Instances.RemoveAt(i);
            else
                Instances[i] = inst;
        }
    }

    public abstract void LoadContent(ContentManager content, GraphicsDevice gd, string pathPrefix);
    public abstract void Update(float deltaSeconds);

    // Default no-ops. Chain layers override Apply; overlay layers override ApplyOverlay.
    public virtual void Apply(SpriteBatch sb, RenderTarget2D source, Viewport viewport) { }
    public virtual void ApplyOverlay(SpriteBatch sb, Viewport viewport) { }

    public abstract void UnloadContent();
}
```

```csharp
using Microsoft.Xna.Framework;
using Microsoft.Xna.Framework.Content;
using Microsoft.Xna.Framework.Graphics;
using ScreenFX.Effects;
using System.Collections.Generic;

namespace ScreenFX;

public sealed class ScreenFXComponent : DrawableGameComponent
{
    private SpriteBatch? _spriteBatch;
    private RenderTarget2D? _ping;
    private RenderTarget2D? _pong;

    private readonly ContentManager _content;
    private readonly string _pathPrefix;

    // Effect layers — concrete types wired in Tasks 3-9.
    // Declared here as the base type so the Draw loop is generic.
    private readonly List<IEffectLayer> _chainLayers = new();
    private readonly List<IEffectLayer> _overlayLayers = new();

    // The game sets this each frame before base.Draw(). Same pattern as WaterRippleBuddy.
    public RenderTarget2D? SceneTarget { get; set; }

    public ScreenFXComponent(Game game, string shaderContentPath = "") : base(game)
    {
        _content    = game.Content;
        _pathPrefix = shaderContentPath;
        DrawOrder   = 100;
    }

    // ── Trigger API ─────────────────────────────────────────────────────────
    // Each trigger converts from screen-space pixels to normalised UV (0-1)
    // before passing to the layer. Layers only know normalised coords.

    public void TriggerForceRipple(Vector2 screenPosition)
    {
        // Wired in Task 3
    }

    public void TriggerGravityWave(float screenY)
    {
        // Wired in Task 4
    }

    public void TriggerScreenShake(float intensity)
    {
        // Wired in Task 5
    }

    public void TriggerChromaticAberration(float intensity, float duration = 0.2f)
    {
        // Wired in Task 6
    }

    public void TriggerHeatHaze(Vector2 screenPosition, float duration = -1f)
    {
        // Wired in Task 7
    }

    public void StopHeatHaze(Vector2 screenPosition)
    {
        // Wired in Task 7
    }

    public void TriggerHitFlash(Color? color = null)
    {
        // Wired in Task 8
    }

    public void TriggerAnimeSuper()
    {
        // Wired in Task 9
    }

    // ── Component lifecycle ──────────────────────────────────────────────────

    protected override void LoadContent()
    {
        var gd = GraphicsDevice;
        _spriteBatch = new SpriteBatch(gd);

        var pp = gd.PresentationParameters;
        _ping = new RenderTarget2D(gd, pp.BackBufferWidth, pp.BackBufferHeight);
        _pong = new RenderTarget2D(gd, pp.BackBufferWidth, pp.BackBufferHeight);

        foreach (var layer in _chainLayers)   layer.LoadContent(_content, gd, _pathPrefix);
        foreach (var layer in _overlayLayers) layer.LoadContent(_content, gd, _pathPrefix);
    }

    protected override void UnloadContent()
    {
        foreach (var layer in _chainLayers)   layer.UnloadContent();
        foreach (var layer in _overlayLayers) layer.UnloadContent();
        _ping?.Dispose();        _ping        = null;
        _pong?.Dispose();        _pong        = null;
        _spriteBatch?.Dispose(); _spriteBatch = null;
    }

    public override void Update(GameTime gameTime)
    {
        float dt = (float)gameTime.ElapsedGameTime.TotalSeconds;
        foreach (var layer in _chainLayers)   layer.Update(dt);
        foreach (var layer in _overlayLayers) layer.Update(dt);
        base.Update(gameTime);
    }

    public override void Draw(GameTime gameTime)
    {
        if (SceneTarget == null || _spriteBatch == null)
        {
            base.Draw(gameTime);
            return;
        }

        var viewport = GraphicsDevice.Viewport;
        RenderTarget2D source = SceneTarget;
        bool usePing = true;

        foreach (var layer in _chainLayers)
        {
            if (!layer.IsActive) continue;
            var dest = usePing ? _ping! : _pong!;
            GraphicsDevice.SetRenderTarget(dest);
            GraphicsDevice.Clear(Color.Transparent);
            layer.Apply(_spriteBatch, source, viewport);
            source   = dest;
            usePing  = !usePing;
        }

        // Blit final result (or unmodified scene if no chain layers ran) to back buffer.
        GraphicsDevice.SetRenderTarget(null);
        _spriteBatch.Begin(SpriteSortMode.Deferred, BlendState.AlphaBlend,
            SamplerState.LinearClamp, DepthStencilState.None, RasterizerState.CullNone);
        _spriteBatch.Draw(source, viewport.Bounds, Color.White);
        _spriteBatch.End();

        // Additive overlay passes run on top of the back buffer.
        foreach (var layer in _overlayLayers)
        {
            if (!layer.IsActive) continue;
            layer.ApplyOverlay(_spriteBatch, viewport);
        }

        base.Draw(gameTime);
    }
}
```

ForceRipple is a circular shockwave — like the WaterRipple effect but tighter and faster. Supports up to 16 simultaneous instances. The shader handles all of them in one pass via `float4 RippleData[16]`.

- [ ] **Step 1: Write the ForceRipple shader**

Create `ScreenFX/Content/ForceRipple.fx`:
```hlsl
#if OPENGL
    #define SV_POSITION POSITION
    #define VS_SHADERMODEL vs_3_0
    #define PS_SHADERMODEL ps_3_0
#else
    #define VS_SHADERMODEL vs_4_0_level_9_1
    #define PS_SHADERMODEL ps_4_0_level_9_1
#endif

#define MAX_RIPPLES 16

float4 RippleData[MAX_RIPPLES]; // xy = normalised screen pos, z = age [0→1], w = unused
float  RippleCount;
float  AspectRatio;
float  WaveSpeed;
float  WaveFrequency;
float  RefractionStrength;
float  ReflectionStrength;
float3 ReflectionColor;

Texture2D SceneTexture;
sampler2D SceneSampler = sampler_state
{
    Texture   = <SceneTexture>;
    AddressU  = Clamp;
    AddressV  = Clamp;
    MagFilter = Linear;
    MinFilter = Linear;
    MipFilter = Linear;
};

struct VertexShaderOutput
{
    float4 Position : SV_POSITION;
    float4 Color    : COLOR0;
    float2 TexCoord : TEXCOORD0;
};

float4 PS(VertexShaderOutput input) : COLOR
{
    float2 uv                = input.TexCoord;
    float2 totalDisplacement = float2(0.0, 0.0);
    float  totalIntensity    = 0.0;
    int    count             = (int)RippleCount;

    for (int i = 0; i < count; i++)
    {
        float2 pos   = RippleData[i].xy;
        float  age   = RippleData[i].z;
        float2 delta = uv - pos;
        delta.x     *= AspectRatio;
        float dist   = length(delta);

        float phase = dist * WaveFrequency - age * WaveSpeed * WaveFrequency;
        // exp(-dist * 8) keeps it as a tight ring; exp(-age * 4) fades over lifetime
        float wave  = sin(phase) * exp(-dist * 8.0) * exp(-age * 4.0);

        float2 dir = float2(0.0, 0.0);
        if (dist > 0.001)
        {
            dir   = normalize(delta);
            dir.x /= AspectRatio;
        }
        totalDisplacement += dir * wave;
        totalIntensity    += abs(wave);
    }

    float2 refractedUV = clamp(uv + totalDisplacement * RefractionStrength, 0.0, 1.0);
    float4 sceneColor  = tex2D(SceneSampler, refractedUV);
    float  reflFactor  = saturate(totalIntensity * ReflectionStrength);
    float3 finalRGB    = lerp(sceneColor.rgb, ReflectionColor, reflFactor);
    return float4(finalRGB, sceneColor.a) * input.Color;
}

technique ForceRipple
{
    pass P0
    {
        PixelShader = compile PS_SHADERMODEL PS();
    }
}
```

- [ ] **Step 1: Write the GravityWave shader**

```hlsl
#if OPENGL
    #define SV_POSITION POSITION
    #define VS_SHADERMODEL vs_3_0
    #define PS_SHADERMODEL ps_3_0
#else
    #define VS_SHADERMODEL vs_4_0_level_9_1
    #define PS_SHADERMODEL ps_4_0_level_9_1
#endif

#define MAX_WAVES 8

float4 WaveData[MAX_WAVES]; // x = normalised Y of ground contact, y = age [0→1], zw = unused
float  WaveCount;
float  WaveSpeed;
float  WaveFrequency;
float  RefractionStrength;

Texture2D SceneTexture;
sampler2D SceneSampler = sampler_state
{
    Texture   = <SceneTexture>;
    AddressU  = Clamp;
    AddressV  = Clamp;
    MagFilter = Linear;
    MinFilter = Linear;
    MipFilter = Linear;
};

struct VertexShaderOutput
{
    float4 Position : SV_POSITION;
    float4 Color    : COLOR0;
    float2 TexCoord : TEXCOORD0;
};

float4 PS(VertexShaderOutput input) : COLOR

    float  totalDispX = 0.0;
    int    count      = (int)WaveCount;

    for (int i = 0; i < count; i++)
    {
        float normY = WaveData[i].x;
        float age   = WaveData[i].y;

        float phase = dy * WaveFrequency - age * WaveSpeed * WaveFrequency;

        float wave  = sin(phase) * exp(-dy * 10.0) * exp(-age * 6.0);
        totalDispX += wave;
    }

    float2 sampledUV = float2(clamp(uv.x + totalDispX * RefractionStrength, 0.0, 1.0), uv.y);
    return tex2D(SceneSampler, sampledUV) * input.Color;
}
```

```hlsl
#endif

float Intensity;    // 0→1, decayed by the layer each frame
float AspectRatio;

Texture2D SceneTexture;
sampler2D SceneSampler = sampler_state
{
    Texture   = <SceneTexture>;
    AddressU  = Clamp;
    AddressV  = Clamp;
    MagFilter = Linear;
    MinFilter = Linear;
    MipFilter = Linear;

    float4 Color    : COLOR0;
    float2 TexCoord : TEXCOORD0;

    float g = tex2D(SceneSampler, uv).g;
    float b = tex2D(SceneSampler, clamp(uv - offset,         0.0, 1.0)).b;
    float a = tex2D(SceneSampler, uv).a;

    return float4(r, g, b, a) * input.Color;
}
```

Up to 8 localised distortion zones. Each zone has a position, radius, and phase offset. Distortion is a continuous sin wave within the zone radius. Zones can be infinite (`duration = -1`) or time-limited. `StopHeatHaze` removes the nearest zone.

- [ ] **Step 1: Write the HeatHaze shader**

```hlsl
#if OPENGL
    #define SV_POSITION POSITION
    #define VS_SHADERMODEL vs_3_0
    #define PS_SHADERMODEL ps_3_0
#else
    #define VS_SHADERMODEL vs_4_0_level_9_1
    #define PS_SHADERMODEL ps_4_0_level_9_1
#endif

#define MAX_ZONES 8

// xy = normalised screen pos, z = radius in UV space, w = phase offset
float4 HazeData[MAX_ZONES];
float  HazeCount;
float  Time;        // total elapsed seconds, for animation
float  AspectRatio;

Texture2D SceneTexture;
sampler2D SceneSampler = sampler_state
{
    Texture   = <SceneTexture>;
    AddressU  = Clamp;
    AddressV  = Clamp;
    MagFilter = Linear;
    MinFilter = Linear;
    MipFilter = Linear;
};

struct VertexShaderOutput
{
    float4 Position : SV_POSITION;
    float4 Color    : COLOR0;
    float2 TexCoord : TEXCOORD0;
};

float4 PS(VertexShaderOutput input) : COLOR
{
    float2 uv        = input.TexCoord;
    float2 totalDisp = float2(0.0, 0.0);
    int    count     = (int)HazeCount;

    for (int i = 0; i < count; i++)
    {
        float2 pos    = HazeData[i].xy;
        float  radius = HazeData[i].z;
        float  phase  = HazeData[i].w;

        float2 delta = uv - pos;
        delta.x     *= AspectRatio;
        float dist   = length(delta);

        if (dist < radius)
        {
            float falloff = 1.0 - (dist / radius);
            // Horizontal shimmer driven by vertical position + time
            float wave = sin(uv.y * 40.0 + Time * 3.0 + phase) * 0.005 * falloff;
            totalDisp.x += wave;
        }
    }

    float2 sampledUV = clamp(uv + totalDisp, 0.0, 1.0);
    return tex2D(SceneSampler, sampledUV) * input.Color;
}
```

```csharp
using Microsoft.Xna.Framework;
using Microsoft.Xna.Framework.Content;
using Microsoft.Xna.Framework.Graphics;
using System;
using System.Collections.Generic;

namespace ScreenFX.Effects;

public sealed class HeatHazeLayer : EffectLayerBase
{
    private struct HazeZone
    {
        public Vector2 Position;    // normalised screen space
        public float   Radius;      // UV space, e.g. 0.15
        public float   PhaseOffset; // random, set once on creation
        public float   Duration;    // -1 = infinite
        public float   Elapsed;
    }

    private Effect?          _effect;
    private EffectParameter? _paramHazeData;
    private EffectParameter? _paramHazeCount;
    private EffectParameter? _paramTime;
    private EffectParameter? _paramAspectRatio;
    private EffectParameter? _paramSceneTexture;

    private readonly List<HazeZone> _zones = new();
    private readonly Vector4[]      _hazeBuffer = new Vector4[8];
    private float _time;
    private readonly Random _rng = new();

    public float ZoneRadius { get; set; } = 0.15f;

    public override bool IsActive => _zones.Count > 0;

    public void Trigger(Vector2 normalizedPosition, float duration = -1f)
    {
        if (_zones.Count >= 8) return;
        _zones.Add(new HazeZone
        {
            Position    = normalizedPosition,
            Radius      = ZoneRadius,
            PhaseOffset = (float)(_rng.NextDouble() * Math.PI * 2.0),
            Duration    = duration,
            Elapsed     = 0f
        });
    }

    public void Stop(Vector2 normalizedPosition)
    {
        if (_zones.Count == 0) return;
        int   closest  = 0;
        float bestDist = float.MaxValue;
        for (int i = 0; i < _zones.Count; i++)
        {
            float d = Vector2.Distance(_zones[i].Position, normalizedPosition);
            if (d < bestDist) { bestDist = d; closest = i; }
        }
        _zones.RemoveAt(closest);
    }

    public override void LoadContent(ContentManager content, GraphicsDevice gd, string pathPrefix)
    {
        _effect           = content.Load<Effect>(pathPrefix + "HeatHaze");
        _paramHazeData    = _effect.Parameters["HazeData"];
        _paramHazeCount   = _effect.Parameters["HazeCount"];
        _paramTime        = _effect.Parameters["Time"];
        _paramAspectRatio = _effect.Parameters["AspectRatio"];
        _paramSceneTexture = _effect.Parameters["SceneTexture"];
    }

    public override void Update(float deltaSeconds)
    {
        _time += deltaSeconds;
        for (int i = _zones.Count - 1; i >= 0; i--)
        {
            var zone = _zones[i];
            if (zone.Duration < 0) continue; // infinite zone
            zone.Elapsed += deltaSeconds;
            if (zone.Elapsed >= zone.Duration)
                _zones.RemoveAt(i);
            else
                _zones[i] = zone;
        }
    }

    public override void Apply(SpriteBatch sb, RenderTarget2D source, Viewport viewport)
    {
        int count = _zones.Count;
        for (int i = 0; i < count; i++)
        {
            var z          = _zones[i];
            _hazeBuffer[i] = new Vector4(z.Position.X, z.Position.Y, z.Radius, z.PhaseOffset);
        }

        _paramHazeData?.SetValue(_hazeBuffer);
        _paramHazeCount?.SetValue((float)count);
        _paramTime?.SetValue(_time);
        _paramAspectRatio?.SetValue((float)viewport.Width / viewport.Height);
        _paramSceneTexture?.SetValue(source);

        sb.Begin(SpriteSortMode.Immediate, BlendState.AlphaBlend,
            SamplerState.LinearClamp, DepthStencilState.None, RasterizerState.CullNone, _effect);
        sb.Draw(source, viewport.Bounds, Color.White);
        sb.End();
    }

        _effect = null;
```

## Task 9: AnimeSuper Overlay Effect

**Files:**
- Create: `ScreenFX/Content/AnimeSuper.fx`
- Create: `ScreenFX/Effects/AnimeSuperLayer.cs`
- Modify: `ScreenFX/ScreenFXComponent.cs`

Timed full-screen sequence driven by a `Progress` float [0→1] over 1.5s:
- 0.0–0.1: white flash (ramps to 1, drops)
- 0.1–0.8: radial speed lines + red vignette
- 0.8–1.0: fade out

A second trigger while active restarts from 0. Only one instance at a time.

- [ ] **Step 1: Write the AnimeSuper shader**

Create `ScreenFX/Content/AnimeSuper.fx`:
```hlsl
#if OPENGL
    #define SV_POSITION POSITION
    #define VS_SHADERMODEL vs_3_0
    #define PS_SHADERMODEL ps_3_0
#else
    #define VS_SHADERMODEL vs_4_0_level_9_1
    #define PS_SHADERMODEL ps_4_0_level_9_1
#endif

float  Progress;       // 0→1 over the full 1.5s sequence
float3 LineColor;      // speed line color (default white)
float3 VignetteColor;  // vignette color (default dark red)

struct VertexShaderOutput
{
    float4 Position : SV_POSITION;
    float4 Color    : COLOR0;
    float2 TexCoord : TEXCOORD0;
};

// frac(x) returns fractional part — used to create repeating stripe pattern.
float4 PS(VertexShaderOutput input) : COLOR
{
    float2 uv       = input.TexCoord;
    float2 centered = uv - float2(0.5, 0.5);
    float  radius   = length(centered);

    // ── White flash (Progress 0.0–0.1) ──────────────────────────────────────
    float flashIntensity = 0.0;
    if (Progress < 0.1)
    {
        float t          = Progress / 0.1;
        flashIntensity   = (t < 0.5) ? t * 2.0 : (1.0 - t) * 2.0;
    }

    // ── Speed lines (Progress 0.1–0.8) ──────────────────────────────────────
    float lineIntensity = 0.0;
    if (Progress >= 0.1 && Progress < 0.8)
    {
        float t = (Progress - 0.1) / 0.7;

        // Polar angle → stripe pattern. 24 lines rotating slowly.
        float angle       = atan2(centered.y, centered.x); // -PI to PI
        float linePattern = frac((angle / (3.14159265 * 2.0)) * 24.0 + t * 0.3);
        float stripe      = step(0.5, linePattern);

        // Lines fade toward screen edge and with time.
        float radFalloff  = max(0.0, 1.0 - radius * 2.5);
        lineIntensity     = stripe * radFalloff * (1.0 - t * 0.4);
    }

    // ── Vignette (Progress 0.1–0.8) ─────────────────────────────────────────
    float vignetteIntensity = 0.0;
    if (Progress >= 0.1 && Progress < 0.8)
    {
        float t            = (Progress - 0.1) / 0.7;
        float vignette     = smoothstep(0.3, 0.65, radius);
        vignetteIntensity  = vignette * (0.5 + 0.3 * sin(t * 3.14159265 * 4.0));
    }

    // ── Fade out (Progress 0.8–1.0) ─────────────────────────────────────────
    float fadeOut = 1.0;
    if (Progress >= 0.8)
        fadeOut = 1.0 - (Progress - 0.8) / 0.2;

    float4 result = float4(0.0, 0.0, 0.0, 0.0);
    result += float4(LineColor,    1.0) * flashIntensity;
    result += float4(LineColor,    1.0) * lineIntensity;
    result += float4(VignetteColor, 1.0) * vignetteIntensity;
    result  = result * fadeOut;

    return result * input.Color;
}

technique AnimeSuper
{
    pass P0
    {
        PixelShader = compile PS_SHADERMODEL PS();
    }
}
```

```csharp
using Microsoft.Xna.Framework;
using Microsoft.Xna.Framework.Graphics;
using Microsoft.Xna.Framework.Input;
using ScreenFX;

namespace ScreenFX.Example;

public class Game1 : Game
{
    private const int ScreenWidth  = 1280;
    private const int ScreenHeight = 720;

    private readonly GraphicsDeviceManager _graphics;
    private SpriteBatch?                   _spriteBatch;
    private Texture2D?                     _background;
    private RenderTarget2D?                _sceneTarget;
    private ScreenFXComponent?             _fx;

    private KeyboardState _prevKeyboard;
    private MouseState    _prevMouse;
    private bool          _heatHazeActive;
    private Vector2       _heatHazePos;

    public Game1()
    {
        _graphics = new GraphicsDeviceManager(this)
        {
            PreferredBackBufferWidth  = ScreenWidth,
            PreferredBackBufferHeight = ScreenHeight
        };
        Content.RootDirectory = "Content";
        IsMouseVisible        = true;
    }

    protected override void Initialize()
    {
        _fx = new ScreenFXComponent(this);
        Components.Add(_fx);
        base.Initialize();
    }

    protected override void LoadContent()
    {
        _spriteBatch = new SpriteBatch(GraphicsDevice);
        _background  = Content.Load<Texture2D>("Braid_screenshot8");
        _sceneTarget = new RenderTarget2D(GraphicsDevice, ScreenWidth, ScreenHeight);
    }

    protected override void Update(GameTime gameTime)
    {
        var kb    = Keyboard.GetState();
        var mouse = Mouse.GetState();

        if (GamePad.GetState(PlayerIndex.One).Buttons.Back == ButtonState.Pressed ||
            kb.IsKeyDown(Keys.Escape))
            Exit();

        // Left click → Force Ripple at cursor
        if (_prevMouse.LeftButton == ButtonState.Released &&
            mouse.LeftButton      == ButtonState.Pressed)
        {
            _fx!.TriggerForceRipple(new Vector2(mouse.X, mouse.Y));
        }

        // Space → Gravity Wave at bottom-centre
        if (_prevKeyboard.IsKeyUp(Keys.Space) && kb.IsKeyDown(Keys.Space))
            _fx!.TriggerGravityWave(ScreenHeight * 0.9f);

        // S → Screen Shake
        if (_prevKeyboard.IsKeyUp(Keys.S) && kb.IsKeyDown(Keys.S))
            _fx!.TriggerScreenShake(0.8f);

        // C → Chromatic Aberration
        if (_prevKeyboard.IsKeyUp(Keys.C) && kb.IsKeyDown(Keys.C))
            _fx!.TriggerChromaticAberration(1.0f, 0.5f);

        // H → Heat Haze toggle at cursor
        if (_prevKeyboard.IsKeyUp(Keys.H) && kb.IsKeyDown(Keys.H))
        {
            if (_heatHazeActive)
            {
                _fx!.StopHeatHaze(_heatHazePos);
                _heatHazeActive = false;
            }
            else
            {
                _heatHazePos    = new Vector2(mouse.X, mouse.Y);
                _fx!.TriggerHeatHaze(_heatHazePos);
                _heatHazeActive = true;
            }
        }

        // F → Hit Flash
        if (_prevKeyboard.IsKeyUp(Keys.F) && kb.IsKeyDown(Keys.F))
            _fx!.TriggerHitFlash();

        // A → Anime Super
        if (_prevKeyboard.IsKeyUp(Keys.A) && kb.IsKeyDown(Keys.A))
            _fx!.TriggerAnimeSuper();

        _prevKeyboard = kb;
        _prevMouse    = mouse;
        base.Update(gameTime);
    }

    protected override void Draw(GameTime gameTime)
    {
        // 1. Render scene to off-screen target
        GraphicsDevice.SetRenderTarget(_sceneTarget);
        GraphicsDevice.Clear(Color.Black);
        _spriteBatch!.Begin();
        _spriteBatch.Draw(_background!, new Rectangle(0, 0, ScreenWidth, ScreenHeight), Color.White);
        _spriteBatch.End();

        // 2. Hand scene target to the component; switch back to back buffer
        _fx!.SceneTarget = _sceneTarget;
        GraphicsDevice.SetRenderTarget(null);

        // 3. Component runs the full effect chain and outputs to back buffer
        base.Draw(gameTime);
    }

    protected override void UnloadContent()
    {
        _sceneTarget?.Dispose();
        base.UnloadContent();
    }
}
```
